begin
declare x varchar(10);
declare y varchar(10);
declare finished result int default true;
declare c1 cursor fro EXIST from tempp;
declare comtinue handler fornot found set finished=false;
open c1;
cursor_c1_loop : loop
fetch c1 into x,y;
if finished = true then 
leave cursor_c1_loop;
endif;
set result= x exist y
insert result into tempp;
close c1;
end; //
